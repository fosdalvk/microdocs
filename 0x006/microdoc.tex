\section{General Theorems}

\subsection{Relative Primes}

\begin{axiom}
$n$ and $m$ are \emph{relative primes} if their only common divisor is $1$.
\end{axiom}

\begin{theorem}
if $a$ and $n$ are relatively prime, then there exists a unique number $i \mod n \suchthat ab \equiv 1 \mod n$
\end{theorem}
